20 #include "Nextion.h"

22 /*
23  * Declare a dual state button object [page id:0,component id:1, component name: "bt0"].
24  */
25 
26 NexDSButton bt0(0, 1, "bt0");
27 
28 NexText t0(0, 2, "t0");
29 
30 
31 char buffer[100] = {0};
32 
33 /*
34  * Register a dual state button object to the touch event list.
35  */
36 NexTouch *nex_listen_list[] =
37 {
38  &bt0,
39  NULL
40 };
41 
42 /*
43  * Dual state button component pop callback function.
44  * In this example,the button's text value will plus one every time when it is released.
45  */
46 void bt0PopCallback(void *ptr)
47 {
48  uint32_t dual_state;
49  NexDSButton *btn = (NexDSButton *)ptr;
50  dbSerialPrintln("b0PopCallback");
51  dbSerialPrint("ptr=");
52  dbSerialPrintln((uint32_t)ptr);
53  memset(buffer, 0, sizeof(buffer));
54 
55  /* Get the state value of dual state button component . */
56  bt0.getValue(&dual_state);
57  if(dual_state)
58  {
59  t0.setText("HI! OPEN STATE");
60  }
61  else
62  {
63  t0.setText("HI! OFF STATE");
64  }
65 }
66 
67 void setup(void)
68 {
69  /* Set the baudrate which is for debug and communicate with Nextion screen. */
70  nexInit();
71 
72  /* Register the pop event callback function of the dual state button component. */
73  bt0.attachPop(bt0PopCallback, &bt0);
74 
75  dbSerialPrintln("setup done");
76 }
77 
78 void loop(void)
79 {
80  /*
81  * When a pop or push event occured every time,
82  * the corresponding component[right page id and component id] in touch event list will be asked.
83  */
84  nexLoop(nex_listen_list);
85 }
