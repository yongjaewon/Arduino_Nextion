57 /*
58  * Text component pop callback function.
59  */
60 void t0PopCallback(void *ptr)

62  dbSerialPrintln("t0PopCallback");

66 /*
67  * Button0 component pop callback function.
68  * In this example,the value of the text component will plus one every time when button0 is released.
69  */
70 void b0PopCallback(void *ptr)
71 {
72  uint16_t len;
73  uint16_t number;
74 
75  dbSerialPrintln("b0PopCallback");
76 
77  memset(buffer, 0, sizeof(buffer));
78  len = sizeof(buffer);
79  t0.getText(buffer, len);
80 
81  number = atoi(buffer);
82  number += 1;
83 
84  memset(buffer, 0, sizeof(buffer));
85  itoa(number, buffer, 10);
86 
87  t0.setText(buffer);
88 }

94 void b1PopCallback(void *ptr)

114 void setup(void)
115 {
116  /* Set the baudrate which is for debug and communicate with Nextion screen. */
117  nexInit();
118 
119  /* Register the pop event callback function of the current text component. */
120  t0.attachPop(t0PopCallback);
121 
122  /* Register the pop event callback function of the current button0 component. */
123  b0.attachPop(b0PopCallback);
124 
125  /* Register the pop event callback function of the current button1 component. */
126  b1.attachPop(b1PopCallback);
127 
128  dbSerialPrintln("setup done");
129 }

131 void loop(void)
132 {
133  /*
134  * When a pop or push event occured every time,
135  * the corresponding component[right page id and component id] in touch event list will be asked.
136  */
137  nexLoop(nex_listen_list);
138 }
